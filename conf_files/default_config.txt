server {
    port 8080;
    host 127.0.0.1; #might be redudant
    server_name ay.com;
    index index.html;
    client_body_size 1MB;
    error_page 404.html;
	directory var/www;
    routes {
        route / {
            methods GET DELETE;
            directory var/www/html;
            auto_index on;
			tryfile index.html
        }
        route /all {
            methods GET DELETE;
            directory var/www/html;
            auto_index on;
        }
		route /cgi-bin {
			cgi .py;
			cgi_path var/www/cgi-bin;
		}
        route /uploads {
            methods GET POST DELETE;
			directory var/www/uploads;
            upload_dir var/www/uploads;
			cgi .py;
			cgi_path cgi;
			auto_index on;
			tryfile example1.html
        }
		route /var/www/delete {
			methods GET DELETE;
            directory var/www/delete;
            auto_index on;
		}
		route /try {
			directory var/www/html/try
			tryfile details.html
		}
		route /redir {
			return /all
		}
    }
}
