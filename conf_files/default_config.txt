server {
    port 8080;
    host localhost; #might be redudant
    server_name example.com;
    index index.html;
    error_pages {
        404 errors/404.html;
        500 errors/500.html;
    }
    limits {
        client_body_size 10MB;
    }
    routes {
        route / {
            methods GET POST DELETE;
            directory var/www/html;
            
            cgi .py;
            auto_index on;
        }
        route /uploads {
            methods POST;
            upload_dir var/www/uploads;
        }
    }
}
